// fills the dropdown with systems from the server $(document).ready( () => { $.ajax({ type: "GET", url: "http://localhost:3000/systeme/", success: (data) => { var systeme = data;//jQuery.parseJSON(data); console.log(data); var dropdown = $('#sys-dropdown-suchen'); dropdown.html(''); dropdown.append('\'\''); if (systeme != '') { $.each(systeme, (k, v) => { dropdown.append(''); } ) } } }) //}); });
